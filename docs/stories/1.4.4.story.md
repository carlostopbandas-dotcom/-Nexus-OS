### Exemplo de teste: leadsService.getAll

```typescript
// tests/services/leadsService.test.ts
import { describe, it, expect, vi, beforeEach } from 'vitest'
import { leadsService } from '@/services/leadsService'

const mockLeads = [
  { id: '1', name: 'João', email: 'joao@test.com', status: 'New', deleted_at: null },
]

describe('leadsService', () => {
  beforeEach(() => {
    mockSupabase.order.mockResolvedValue({ data: mockLeads, error: null })
  })

  it('getAll retorna leads sem deleted_at', async () => {
    const { data, error } = await leadsService.getAll()
    expect(error).toBeNull()
    expect(data).toHaveLength(1)
    expect(data![0].name).toBe('João')
  })

  it('getAll retorna error quando Supabase falha', async () => {
    mockSupabase.order.mockResolvedValue({ data: null, error: { message: 'DB error' } })
    const { data, error } = await leadsService.getAll()
    expect(data).toBeNull()
    expect(error).toBe('DB error')
  })
})
```
