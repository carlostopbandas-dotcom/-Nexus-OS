### Lógica de ROAS

```typescript
// Receita: somar store_metrics.total_sales no período
const { data: metrics } = await supabase
  .from('store_metrics')
  .select('total_sales')
  .eq('store_name', store)
  .gte('recorded_at', period.start)
  .lte('recorded_at', period.end)

// Ads Spend: somar ads_campaigns_cache.spend no período
const { data: ads } = await supabase
  .from('ads_campaigns_cache')
  .select('spend')
  .eq('store_name', store)
  .gte('date_start', period.start)

const revenue = metrics?.reduce((sum, m) => sum + m.total_sales, 0) ?? 0
const adSpend = ads?.reduce((sum, a) => sum + a.spend, 0) ?? 0
const adsConfigured = (ads?.length ?? 0) > 0
const roas = adsConfigured && adSpend > 0 ? revenue / adSpend : null
const grossProfit = revenue - adSpend
```

### Estrutura de migration 011

```sql
-- 011_financial_entries.sql
CREATE TABLE IF NOT EXISTS financial_entries (
  id            uuid DEFAULT gen_random_uuid() PRIMARY KEY,
  business_unit text NOT NULL CHECK (business_unit IN (
    'vcchic','moriel','sezo',
    '3d_cend','3d_marketing','3d_producao','3d_inovacao','3d_negocios','3d_csc'
  )),
  entry_type    text NOT NULL CHECK (entry_type IN ('revenue','expense','cmv')),
  category      text,
  amount        numeric(12,2) NOT NULL,
  description   text,
  entry_date    date NOT NULL,
  created_at    timestamptz DEFAULT now(),
  updated_at    timestamptz DEFAULT now()
);

CREATE INDEX IF NOT EXISTS idx_financial_entries_unit_date
  ON financial_entries (business_unit, entry_date);

ALTER TABLE financial_entries ENABLE ROW LEVEL SECURITY;

-- CEO: ALL
CREATE POLICY "ceo_all_financial_entries" ON financial_entries
  FOR ALL USING (get_my_role() = 'ceo');

-- gestor_vcchic: SELECT apenas VcChic, Moriel, Sezo
CREATE POLICY "gestor_vcchic_select_financial_entries" ON financial_entries
  FOR SELECT USING (
    get_my_role() = 'gestor_vcchic'
    AND business_unit IN ('vcchic','moriel','sezo')
  );

-- Rollback:
-- DROP TABLE IF EXISTS financial_entries CASCADE;
```
